172 /* The lowest priority. */
173 #define configKERNEL_INTERRUPT_PRIORITY ( 31 << (8 - configPRIO_BITS) )
174 /* Priority 5, or 160 as only the top three bits are implemented. */
175 /* !!!! configMAX_SYSCALL_INTERRUPT_PRIORITY must not be set to zero !!!!
176 See http://www.FreeRTOS.org/RTOS-Cortex-M3-M4.html. */
177 #define configMAX_SYSCALL_INTERRUPT_PRIORITY ( 5 << (8 - configPRIO_BITS) )
178 
179 /* Priorities passed to NVIC_SetPriority() do not require shifting as the
180 function does the shifting itself. Note these priorities need to be equal to
181 or lower than configMAX_SYSCALL_INTERRUPT_PRIORITY - therefore the numeric
182 value needs to be equal to or greater than 5 (on the Cortex-M3 the lower the
183 numeric value the higher the interrupt priority). */
